!*> Local variables

!define xpathCheckbox {!-//span[text()='@sectionName']/..//label[normalize-space(.)='@checkboxText']-!}
!define xpathTitle {!-//span[text()='@sectionName']/../div[@class='formElement_inputfields'][@rowNumber]//label[text()='@titleText']-!}
!define xpathTextbox {${xpathTitle}!-/following-sibling::input-!}
